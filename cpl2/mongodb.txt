// Connect to MongoDB (assuming MongoDB is running on localhost:27017)
use mydatabase;

// Create (Insert) Operation
db.students.insertMany([
  { name: 'John Doe', age: 25, course: 'Computer Science' },
  { name: 'Alice Smith', age: 28, course: 'Engineering' },
  { name: 'Bob Johnson', age: 23, course: 'Mathematics' }
]);

// Read (Retrieve) Operation
print('All students:');
db.students.find().forEach(printjson);

// Update Operation
print('Updating student Alice Smith...');
db.students.updateOne(
  { name: 'Alice Smith' },
  { $set: { age: 29 } }
);
print('Updated student Alice Smith:');
printjson(db.students.findOne({ name: 'Alice Smith' }));

// Delete Operation
print('Deleting student Bob Johnson...');
db.students.deleteOne({ name: 'Bob Johnson' });
print('Remaining students after deletion:');
db.students.find().forEach(printjson);
